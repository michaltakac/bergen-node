\centerline{\textbf{\Large Matematika 1 - Limita funkcie}}
Úlohy 2\\
\pr Daná je funkcia $f(x)=\dfrac{x^2+4x+3}{x^2-x-2}$. Vypočítajte limitu funkcie $f$ vo vlastných bodoch $-2,-1,0,1,2$ a v nevlastných bodoch $-\infty, \infty$.

\pr Vypočítajte nasledujúce limity 
\begin{enumerate}
\item[a)] $\lim_{x\rightarrow 0}\dfrac{1-\cos 2x}{\sin x \cdot \cos x}$ $=0$
\item[b)] $\lim_{x\rightarrow \frac{\pi}{2}}\dfrac{cotg x - \cos x}{\cos^3 x}$ $=-\sqrt{3}$
\end{enumerate}

\pr Vypočítajte nasledujúce limity 
\begin{enumerate}
\item[a)] $\lim_{x\rightarrow \infty}\dfrac{\sqrt{3x^2+1}}{x}$ $=\sqrt{3}$
\item[b)] $\lim_{x\rightarrow -\infty}\dfrac{\sqrt{3x^2+1}}{x}$ $=-\sqrt{3}$
\end{enumerate}

\pr Vypočítajte nasledujúce limity
 
\begin{enumerate}
\item[a)] $\lim_{x\rightarrow -1}3^{\frac{2}{x+1}}$
\item[b)] $\lim_{x\rightarrow \infty}\left(\frac{3x+1}{x+1}\right)^{x+1}$ $=\infty$
\item[c)] $\lim_{x\rightarrow -\infty}\left(\frac{3x+1}{x+1}\right)^{x+1}$ $=0$
\item[d)] $\lim_{x\rightarrow \infty}\left(\frac{3x+1}{x+1}\right)^{\frac{1}{x+1}}$ $=1$
\end{enumerate}
